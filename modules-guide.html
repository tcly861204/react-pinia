  language: string

      id: 1,

      }
    },
    modules: {
      profile: {
        state: (): ProfileState => ({
          avatar: 'default.png',
          bio: ''

          updateProfile(avatar: string, bio: string) {
            this.avatar = avatar
            this.bio = bio
          }
        }
      },
      settings: {
        state: (): SettingsState => ({
          theme: 'light',
          language: 'zh-CN'
        }),
        actions: {
          toggleTheme() {
            this.theme = this.theme === 'light' ? 'dark' : 'light'
          },
          setLanguage(lang: string) {
            this.language = lang
          }
        }
      }
    }
  }
})
